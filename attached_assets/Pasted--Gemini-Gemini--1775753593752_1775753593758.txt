const axios = require('axios');

setInterval(() => {
  const url = `https://${process.env.REPL_SLUG}.${process.env.REPL_OWNER}.repl.co`;
  axios.get(url).then(() => {
    console.log('Self-ping successful: Keeping the bot alive!');
  }).catch(() => {
    console.log('Self-ping failed, but that is okay.');
  });
}, 240000); // كل 4 دقائق
